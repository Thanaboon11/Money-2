<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=no">
    <title>My Sweet Finance Diary ✨</title>
    <script src="https://cdn.tailwindcss.com"></script>
    <link href="https://fonts.googleapis.com/css2?family=Anuphan:wght@100;400;700&display=swap" rel="stylesheet">
    
    <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
    <script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    <script src="https://cdn.sheetjs.com/xlsx-0.20.0/package/dist/xlsx.full.min.js"></script>
    
    <script src="https://www.gstatic.com/firebasejs/10.7.1/firebase-app-compat.js"></script>
    <script src="https://www.gstatic.com/firebasejs/10.7.1/firebase-database-compat.js"></script>

    <style>
        body { font-family: 'Anuphan', sans-serif; -webkit-tap-highlight-color: transparent; }
        .custom-scrollbar::-webkit-scrollbar { width: 5px; }
        .custom-scrollbar::-webkit-scrollbar-thumb { background: #FFC0CB; border-radius: 10px; }
    </style>
</head>
<body class="bg-[#FFF5F7] min-h-screen pb-20">
    <div id="root"></div>

    <script type="text/babel">
        const { useState, useEffect, useMemo } = React;

        // ข้อมูลการเชื่อมต่อ Firebase ของคุณ Jiraporn Traganthai
       const firebaseConfig = {
  apiKey: "AIzaSyALcH0D-G0H6g0yfgLHrxKcYiHZAq7jAGs",
  authDomain: "monney-33caf.firebaseapp.com",
  databaseURL: "https://monney-33caf-default-rtdb.asia-southeast1.firebasedatabase.app",
  projectId: "monney-33caf",
  storageBucket: "monney-33caf.firebasestorage.app",
  messagingSenderId: "709336346002",
  appId: "1:709336346002:web:444bd02c294bd194d6e54b",
  measurementId: "G-XR98M0L8KP"
};

        if (!firebase.apps.length) firebase.initializeApp(firebaseConfig);
        const db = firebase.database();

        const App = () => {
            const [accounts, setAccounts] = useState([]);
            const [transactions, setTransactions] = useState([]);
            const [activeAcc, setActiveAcc] = useState('all');
            const [editingId, setEditingId] = useState(null);
            const [showAddAcc, setShowAddAcc] = useState(false);
            const [newAccName, setNewAccName] = useState('');
            const [formData, setFormData] = useState({ description: '', amount: '', type: 'expense', accountId: '', date: new Date().toISOString().split('T')[0] });

            useEffect(() => {
                // ดึงข้อมูลบัญชี (Accounts)
                db.ref('accounts').on('value', snap => {
                    const data = snap.val();
                    if (data) {
                        const list = Object.keys(data).map(k => ({id: k, ...data[k]}));
                        setAccounts(list);
                        if (!formData.accountId) setFormData(f => ({...f, accountId: list[0].id}));
                    } else {
                        db.ref('accounts').push({name: 'เงินสด 💵', color: 'from-pink-400 to-rose-300'});
                    }
                });
                // ดึงข้อมูลรายการ (Transactions)
                db.ref('transactions').on('value', snap => {
                    const data = snap.val();
                    if (data) setTransactions(Object.keys(data).map(k => ({id: k, ...data[k]})).reverse());
                    else setTransactions([]);
                });
            }, []);

            // ฟังก์ชันส่งออก Excel แบบแยก Sheet ตามบัญชี ✨
            const exportExcel = () => {
                const wb = XLSX.utils.book_new();

                // 1. Sheet รวมทุกรายการ
                const allData = transactions.map(t => ({
                    'วันที่': t.date, 'รายการ': t.description, 'ประเภท': t.type === 'income' ? 'รับ' : 'จ่าย',
                    'จำนวน': t.amount, 'บัญชี': accounts.find(a => a.id === t.accountId)?.name || 'ทั่วไป'
                }));
                XLSX.utils.book_append_sheet(wb, XLSX.utils.json_to_sheet(allData), "รวมทุกบัญชี");

                // 2. แยก Sheet ตามชื่อบัญชี
                accounts.forEach(acc => {
                    const accTransactions = transactions.filter(t => t.accountId === acc.id);
                    if (accTransactions.length > 0) {
                        const sheetData = accTransactions.map(t => ({
                            'วันที่': t.date, 'รายการ': t.description,
                            'ประเภท': t.type === 'income' ? 'รับ' : 'จ่าย', 'จำนวนเงิน': t.amount
                        }));
                        XLSX.utils.book_append_sheet(wb, XLSX.utils.json_to_sheet(sheetData), acc.name.substring(0, 31));
                    }
                });

                XLSX.writeFile(wb, `Diary_Finance_Report_${new Date().toISOString().slice(0,10)}.xlsx`);
            };

            const handleSubmit = (e) => {
                e.preventDefault();
                if (!formData.description || !formData.amount) return;
                const payload = {...formData, amount: parseFloat(formData.amount)};
                if (editingId) db.ref(`transactions/${editingId}`).update(payload);
                else db.ref('transactions').push(payload);
                setEditingId(null);
                setFormData({...formData, description: '', amount: ''});
            };

            const stats = useMemo(() => {
                const filtered = transactions.filter(t => activeAcc === 'all' || t.accountId === activeAcc);
                return filtered.reduce((acc, curr) => {
                    if (curr.type === 'income') { acc.in += curr.amount; acc.bal += curr.amount; }
                    else { acc.out += curr.amount; acc.bal -= curr.amount; }
                    return acc;
                }, { in: 0, out: 0, bal: 0 });
            }, [transactions, activeAcc]);

            return (
                <div className="max-w-6xl mx-auto px-4 pt-8">
                    <h1 className="text-3xl font-black text-rose-500 text-center mb-8">Diary Finance Cloud ✨</h1>
                    
                    {/* สรุปยอดเงิน */}
                    <div className="grid grid-cols-1 sm:grid-cols-3 gap-4 mb-8">
                        <div className="bg-white p-6 rounded-3xl shadow-sm text-center border-b-4 border-pink-100">
                            <p className="text-2xl font-black">฿{stats.bal.toLocaleString()}</p>
                            <p className="text-[10px] font-bold text-slate-300">ยอดคงเหลือ</p>
                        </div>
                        <div className="bg-emerald-400 p-6 rounded-3xl text-white text-center">
                            <p className="text-2xl font-black">฿{stats.in.toLocaleString()}</p>
                            <p className="text-[10px] font-bold opacity-70">รายรับ</p>
                        </div>
                        <div className="bg-rose-400 p-6 rounded-3xl text-white text-center">
                            <p className="text-2xl font-black">฿{stats.out.toLocaleString()}</p>
                            <p className="text-[10px] font-bold opacity-70">รายจ่าย</p>
                        </div>
                    </div>

                    <div className="grid grid-cols-1 lg:grid-cols-12 gap-6">
                        {/* คอลัมน์ซ้าย: ฟอร์มบันทึก */}
                        <div className="lg:col-span-5 space-y-6">
                            <div className="bg-white p-8 rounded-[2.5rem] shadow-xl border-4 border-white">
                                <h2 className="font-black text-xl mb-6">{editingId ? '🪄 แก้ไขรายการ' : '📔 จดรายการใหม่'}</h2>
                                <form onSubmit={handleSubmit} className="space-y-4">
                                    <div className="flex bg-slate-50 p-1.5 rounded-2xl">
                                        <button type="button" onClick={() => setFormData({...formData, type: 'income'})} className={`flex-1 py-3 rounded-xl font-bold text-sm ${formData.type === 'income' ? 'bg-white shadow-md text-emerald-500' : 'text-slate-400'}`}>รายรับ</button>
                                        <button type="button" onClick={() => setFormData({...formData, type: 'expense'})} className={`flex-1 py-3 rounded-xl font-bold text-sm ${formData.type === 'expense' ? 'bg-white shadow-md text-rose-500' : 'text-slate-400'}`}>รายจ่าย</button>
                                    </div>
                                    <select value={formData.accountId} onChange={e => setFormData({...formData, accountId: e.target.value})} className="w-full p-4 bg-slate-50 rounded-2xl font-bold outline-none border-none">
                                        {accounts.map(a => <option key={a.id} value={a.id}>{a.name}</option>)}
                                    </select>
                                    <input type="date" value={formData.date} onChange={e => setFormData({...formData, date: e.target.value})} className="w-full p-4 bg-slate-50 rounded-2xl font-bold outline-none border-none text-slate-500" />
                                    <input type="text" placeholder="รายการ..." value={formData.description} onChange={e => setFormData({...formData, description: e.target.value})} className="w-full p-4 bg-slate-50 rounded-2xl font-bold outline-none border-none shadow-inner" />
                                    <input type="number" placeholder="0.00" value={formData.amount} onChange={e => setFormData({...formData, amount: e.target.value})} className="w-full p-5 bg-slate-50 rounded-2xl font-black text-3xl text-center outline-none border-none shadow-inner" />
                                    <button type="submit" className="w-full py-5 bg-pink-500 text-white font-black rounded-3xl shadow-lg hover:scale-95 transition-all">บันทึกข้อมูล ✨</button>
                                    <button type="button" onClick={exportExcel} className="w-full py-3 border-2 border-emerald-400 text-emerald-600 font-black rounded-3xl text-sm hover:bg-emerald-50 transition-all">ส่งออก EXCEL แยก Sheet 📑</button>
                                </form>
                            </div>
                        </div>

                        {/* คอลัมน์ขวา: ประวัติ & จัดการบัญชี */}
                        <div className="lg:col-span-7 space-y-6">
                            <div className="bg-white/40 backdrop-blur-md p-8 rounded-[2.5rem] border-4 border-white shadow-sm min-h-[400px]">
                                <div className="flex justify-between items-center mb-6">
                                    <h2 className="font-black text-xl text-slate-700">กระเป๋าเงิน 👛</h2>
                                    <button onClick={() => setShowAddAcc(true)} className="w-8 h-8 bg-pink-500 text-white rounded-xl font-bold">+</button>
                                </div>
                                <div className="flex gap-2 overflow-x-auto pb-4 mb-6">
                                    <button onClick={() => setActiveAcc('all')} className={`px-4 py-2 rounded-xl text-[10px] font-bold whitespace-nowrap ${activeAcc === 'all' ? 'bg-pink-500 text-white' : 'bg-white text-pink-400'}`}>ทั้งหมด</button>
                                    {accounts.map(a => (
                                        <div key={a.id} className="relative group">
                                            <button onClick={() => setActiveAcc(a.id)} className={`px-4 py-2 rounded-xl text-[10px] font-bold whitespace-nowrap ${activeAcc === a.id ? 'bg-white shadow-sm' : 'bg-white/30 text-slate-400'}`}>{a.name}</button>
                                            <button onClick={() => {if(confirm('ลบบัญชีนี้?')) db.ref(`accounts/${a.id}`).remove()}} className="absolute -top-1 -right-1 w-4 h-4 bg-rose-500 text-white rounded-full text-[8px] opacity-0 group-hover:opacity-100 transition-all flex items-center justify-center">✕</button>
                                        </div>
                                    ))}
                                </div>
                                <div className="space-y-3">
                                    {transactions.filter(t => activeAcc === 'all' || t.accountId === activeAcc).map(t => (
                                        <div key={t.id} className="p-4 bg-white/90 rounded-2xl shadow-sm flex justify-between items-center group">
                                            <div>
                                                <p className="font-bold text-slate-700">{t.description}</p>
                                                <p className="text-[10px] text-pink-400 font-bold uppercase">{t.date} ● {accounts.find(a => a.id === t.accountId)?.name}</p>
                                            </div>
                                            <div className="flex items-center gap-3">
                                                <p className={`font-black ${t.type === 'income' ? 'text-emerald-500' : 'text-slate-700'}`}>฿{t.amount.toLocaleString()}</p>
                                                <div className="flex gap-1 md:opacity-0 group-hover:opacity-100 transition-all">
                                                    <button onClick={() => {setEditingId(t.id); setFormData(t); window.scrollTo({top:0, behavior:'smooth'})}} className="p-2 text-indigo-400">✎</button>
                                                    <button onClick={() => {if(confirm('ลบรายการ?')) db.ref(`transactions/${t.id}`).remove()}} className="p-2 text-rose-400">🗑</button>
                                                </div>
                                            </div>
                                        </div>
                                    ))}
                                </div>
                            </div>
                        </div>
                    </div>

                    {/* Modal เพิ่มบัญชี */}
                    {showAddAcc && (
                        <div className="fixed inset-0 z-50 flex items-center justify-center bg-rose-200/50 backdrop-blur-md p-6">
                            <div className="bg-white w-full max-w-sm p-12 rounded-[3.5rem] shadow-2xl border-8 border-white animate-in zoom-in-95 duration-300 text-center">
                                <h3 className="text-3xl font-black text-slate-700 mb-8">เพิ่มกระเป๋าจ้า 🌈</h3>
                                <input autoFocus type="text" value={newAccName} onChange={e => setNewAccName(e.target.value)} placeholder="ชื่อบัญชี..." className="w-full px-8 py-5 bg-slate-50 rounded-[2rem] outline-none font-bold text-center mb-10" />
                                <div className="flex gap-4">
                                    <button onClick={() => setShowAddAcc(false)} className="flex-1 py-4 text-slate-300 font-bold">ยกเลิก</button>
                                    <button onClick={() => {if(newAccName){db.ref('accounts').push({name: newAccName, color: 'from-pink-400'}); setNewAccName(''); setShowAddAcc(false);}}} className="flex-[2] py-4 bg-pink-500 text-white font-black rounded-[2rem]">สร้างเลย!</button>
                                </div>
                            </div>
                        </div>
                    )}
                </div>
            );
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
</body>
