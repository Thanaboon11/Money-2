    <meta name="viewport" content="width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=no">

    <script src="https://cdn.tailwindcss.com"></script>
    <link href="https://fonts.googleapis.com/css2?family=Anuphan:wght@100;400;700&display=swap" rel="stylesheet">
    
    <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
    <script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    <script src="https://cdn.sheetjs.com/xlsx-0.20.0/package/dist/xlsx.full.min.js"></script>
    
    <script src="https://www.gstatic.com/firebasejs/10.7.1/firebase-app-compat.js"></script>
    <script src="https://www.gstatic.com/firebasejs/10.7.1/firebase-database-compat.js"></script>

    <style>
        body { font-family: 'Anuphan', sans-serif; -webkit-tap-highlight-color: transparent; }
        .custom-scrollbar::-webkit-scrollbar { width: 5px; }
        .custom-scrollbar::-webkit-scrollbar-thumb { background: #FFC0CB; border-radius: 10px; }
        .no-scrollbar::-webkit-scrollbar { display: none; }
    </style>
</head>
<body class="bg-[#FFF5F7] min-h-screen pb-20">
    <div id="root"></div>

    <script type="text/babel">
        const { useState, useEffect, useMemo } = React;

 const firebaseConfig = {
  apiKey: "AIzaSyALcH0D-G0H6g0yfgLHrxKcYiHZAq7jAGs",
  authDomain: "monney-33caf.firebaseapp.com",
  databaseURL: "https://monney-33caf-default-rtdb.asia-southeast1.firebasedatabase.app",
  projectId: "monney-33caf",
  storageBucket: "monney-33caf.firebasestorage.app",
  messagingSenderId: "709336346002",
  appId: "1:709336346002:web:444bd02c294bd194d6e54b",
  measurementId: "G-XR98M0L8KP"
};


        if (!firebase.apps.length) firebase.initializeApp(firebaseConfig);
        const db = firebase.database();

        const App = () => {
            const [accounts, setAccounts] = useState([]);
            const [transactions, setTransactions] = useState([]);
            const [activeAcc, setActiveAcc] = useState('all');
            const [editingId, setEditingId] = useState(null);
            const [showAddAcc, setShowAddAcc] = useState(false);
            const [newAccName, setNewAccName] = useState('');
            const [formData, setFormData] = useState({ description: '', amount: '', type: 'expense', accountId: '', date: new Date().toISOString().split('T')[0] });

            useEffect(() => {
                db.ref('accounts').on('value', snap => {
                    const data = snap.val();
                    if (data) {
                        const list = Object.keys(data).map(k => ({id: k, ...data[k]}));
                        setAccounts(list);
                        if (!formData.accountId) setFormData(f => ({...f, accountId: list[0].id}));
                    } else {
                        db.ref('accounts').push({name: 'เงินสด 💵', color: 'from-pink-400 to-rose-300'});
                    }
                });
                db.ref('transactions').on('value', snap => {
                    const data = snap.val();
                    if (data) setTransactions(Object.keys(data).map(k => ({id: k, ...data[k]})).reverse());
                    else setTransactions([]);
                });
            }, []);

            const monthlySummary = useMemo(() => {
                const summary = {};
                transactions.forEach(t => {
                    const monthYear = t.date.substring(0, 7);
                    if (!summary[monthYear]) summary[monthYear] = { in: 0, out: 0 };
                    if (t.type === 'income') summary[monthYear].in += t.amount;
                    else summary[monthYear].out += t.amount;
                });
                return Object.entries(summary).sort((a, b) => b[0].localeCompare(a[0]));
            }, [transactions]);

            const exportExcel = () => {
                const data = transactions.map(t => ({
                    'วันที่': t.date, 'รายการ': t.description, 
                    'ประเภท': t.type === 'income' ? 'รับ' : 'จ่าย', 'จำนวน': t.amount,
                    'บัญชี': accounts.find(a => a.id === t.accountId)?.name
                }));
                const ws = XLSX.utils.json_to_sheet(data);
                const wb = XLSX.utils.book_new();
                XLSX.utils.book_append_sheet(wb, ws, "Financials");
                XLSX.writeFile(wb, "Diary_Finance.xlsx");
            };

            const handleAddAccount = () => {
                if (!newAccName.trim()) return;
                db.ref('accounts').push({ name: newAccName, color: 'from-pink-400 to-rose-300' });
                setNewAccName('');
                setShowAddAcc(false);
            };

            const handleSubmit = (e) => {
                e.preventDefault();
                if (!formData.description || !formData.amount) return;
                const payload = {...formData, amount: parseFloat(formData.amount)};
                if (editingId) db.ref(`transactions/${editingId}`).update(payload);
                else db.ref('transactions').push(payload);
                setEditingId(null);
                setFormData({...formData, description: '', amount: ''});
            };

            const currentStats = useMemo(() => {
                const filtered = transactions.filter(t => activeAcc === 'all' || t.accountId === activeAcc);
                return filtered.reduce((acc, curr) => {
                    if (curr.type === 'income') { acc.in += curr.amount; acc.bal += curr.amount; }
                    else { acc.out += curr.amount; acc.bal -= curr.amount; }
                    return acc;
                }, { in: 0, out: 0, bal: 0 });
            }, [transactions, activeAcc]);

            return (
                <div className="max-w-6xl mx-auto px-4 pt-8">
                    <header className="text-center mb-8">
                        <h1 className="text-3xl font-black text-rose-500 mb-1">Diary Finance ✨</h1>
                        <p className="text-slate-400 text-sm font-bold italic">เพิ่มบัญชีและสรุปยอดได้ครบจ้า</p>
                    </header>
                    
                    <div className="grid grid-cols-1 sm:grid-cols-3 gap-4 mb-8">
                        <div className="bg-white p-6 rounded-3xl shadow-sm text-center border-b-4 border-pink-100">
                            <p className="text-2xl font-black text-slate-800">฿{currentStats.bal.toLocaleString()}</p>
                            <p className="text-[10px] font-bold text-slate-300 uppercase mt-1">ยอดคงเหลือรวม</p>
                        </div>
                        <div className="bg-emerald-400 p-6 rounded-3xl shadow-sm text-center text-white">
                            <p className="text-2xl font-black">฿{currentStats.in.toLocaleString()}</p>
                            <p className="text-[10px] font-bold text-emerald-100 uppercase mt-1">รายรับ</p>
                        </div>
                        <div className="bg-rose-400 p-6 rounded-3xl shadow-sm text-center text-white">
                            <p className="text-2xl font-black">฿{currentStats.out.toLocaleString()}</p>
                            <p className="text-[10px] font-bold text-rose-100 uppercase mt-1">รายจ่าย</p>
                        </div>
                    </div>

                    <div className="grid grid-cols-1 lg:grid-cols-12 gap-6 items-start">
                        <div className="lg:col-span-5 space-y-6">
                            <div className="bg-white p-6 sm:p-8 rounded-[2.5rem] shadow-xl border-4 border-white">
                                <h2 className="font-black text-xl mb-6 text-slate-700">{editingId ? '🪄 แก้ไขรายการ' : '📔 จดรายการใหม่'}</h2>
                                <form onSubmit={handleSubmit} className="space-y-4">
                                    <div className="flex bg-slate-50 p-1.5 rounded-2xl">
                                        <button type="button" onClick={() => setFormData({...formData, type: 'income'})} className={`flex-1 py-3 rounded-xl font-bold text-sm transition-all ${formData.type === 'income' ? 'bg-white shadow-md text-emerald-500' : 'text-slate-400'}`}>รายรับ 🍰</button>
                                        <button type="button" onClick={() => setFormData({...formData, type: 'expense'})} className={`flex-1 py-3 rounded-xl font-bold text-sm transition-all ${formData.type === 'expense' ? 'bg-white shadow-md text-rose-500' : 'text-slate-400'}`}>รายจ่าย 🎈</button>
                                    </div>
                                    <div className="grid grid-cols-2 gap-3">
                                        <select value={formData.accountId} onChange={e => setFormData({...formData, accountId: e.target.value})} className="p-4 bg-slate-50 rounded-2xl font-bold text-sm outline-none">
                                            {accounts.map(a => <option key={a.id} value={a.id}>{a.name}</option>)}
                                        </select>
                                        <input type="date" value={formData.date} onChange={e => setFormData({...formData, date: e.target.value})} className="p-4 bg-slate-50 rounded-2xl font-bold text-sm outline-none text-slate-500" />
                                    </div>
                                    <input type="text" placeholder="รายการ..." value={formData.description} onChange={e => setFormData({...formData, description: e.target.value})} className="w-full p-4 bg-slate-50 rounded-2xl font-bold outline-none" />
                                    <input type="number" placeholder="0.00" value={formData.amount} onChange={e => setFormData({...formData, amount: e.target.value})} className="w-full p-5 bg-slate-50 rounded-2xl font-black text-3xl text-center outline-none" />
                                    
                                    <button type="submit" className="w-full py-5 bg-pink-500 text-white font-black rounded-3xl shadow-lg active:scale-95 transition-all">บันทึกรายการ ✨</button>
                                    <button type="button" onClick={exportExcel} className="w-full py-3 text-emerald-500 font-bold text-xs border-2 border-emerald-50 rounded-3xl">ส่งออก EXCEL 📑</button>
                                </form>
                            </div>

                            <div className="bg-white/60 p-6 rounded-[2.5rem] border-4 border-white">
                                <h3 className="font-black text-lg mb-4 text-slate-600">📊 สรุปรายเดือน</h3>
                                <div className="space-y-3">
                                    {monthlySummary.map(([month, val]) => (
                                        <div key={month} className="bg-white p-4 rounded-2xl flex justify-between items-center shadow-sm">
                                            <div>
                                                <p className="font-black text-slate-600">{new Date(month).toLocaleDateString('th-TH', { month: 'long', year: 'numeric' })}</p>
                                                <p className="text-[10px] font-bold text-emerald-500">รับ: ฿{val.in.toLocaleString()}</p>
                                            </div>
                                            <div className="text-right">
                                                <p className="font-black text-rose-500">฿{val.out.toLocaleString()}</p>
                                            </div>
                                        </div>
                                    ))}
                                </div>
                            </div>
                        </div>

                        <div className="lg:col-span-7">
                            <div className="bg-white/40 backdrop-blur-md p-6 sm:p-8 rounded-[2.5rem] border-4 border-white shadow-sm min-h-[400px]">
                                <div className="mb-6">
                                    <div className="flex justify-between items-center mb-4">
                                        <h2 className="font-black text-xl text-slate-700">กระเป๋าเงินของฉัน 👛</h2>
                                        <button onClick={() => setShowAddAcc(true)} className="w-8 h-8 bg-pink-500 text-white rounded-xl font-bold flex items-center justify-center shadow-lg">+</button>
                                    </div>
                                    <div className="flex gap-2 overflow-x-auto no-scrollbar pb-2">
                                        <button onClick={() => setActiveAcc('all')} className={`px-4 py-2 rounded-xl text-[10px] font-bold whitespace-nowrap ${activeAcc === 'all' ? 'bg-pink-500 text-white shadow-md' : 'bg-white text-pink-400'}`}>ทั้งหมด</button>
                                        {accounts.map(a => (
                                            <div key={a.id} className="relative group flex-shrink-0">
                                                <button onClick={() => setActiveAcc(a.id)} className={`px-4 py-2 rounded-xl text-[10px] font-bold whitespace-nowrap ${activeAcc === a.id ? 'bg-white shadow-sm ring-1 ring-pink-100' : 'bg-white/50 text-slate-400'}`}>{a.name}</button>
                                                <button onClick={() => {if(confirm('ลบบัญชีนี้?')) db.ref(`accounts/${a.id}`).remove()}} className="absolute -top-1 -right-1 w-4 h-4 bg-rose-400 text-white rounded-full text-[8px] flex items-center justify-center opacity-0 group-hover:opacity-100 transition-all">✕</button>
                                            </div>
                                        ))}
                                    </div>
                                </div>
                                <div className="space-y-3 border-t-2 border-white/50 pt-6">
                                    {transactions.filter(t => activeAcc === 'all' || t.accountId === activeAcc).map(t => (
                                        <div key={t.id} className="p-4 bg-white/90 rounded-2xl shadow-sm flex justify-between items-center group">
                                            <div className="flex items-center gap-4">
                                                <div className={`w-10 h-10 rounded-xl flex items-center justify-center text-lg ${t.type === 'income' ? 'bg-emerald-50 text-emerald-500' : 'bg-rose-50 text-rose-500'}`}>{t.type === 'income' ? '🍰' : '🎈'}</div>
                                                <div>
                                                    <p className="font-bold text-slate-700 text-sm">{t.description}</p>
                                                    <p className="text-[10px] text-slate-300 font-bold uppercase">{t.date} ● {accounts.find(a => a.id === t.accountId)?.name}</p>
                                                </div>
                                            </div>
                                            <div className="flex items-center gap-2">
                                                <p className={`font-black text-base ${t.type === 'income' ? 'text-emerald-500' : 'text-slate-800'}`}>฿{t.amount.toLocaleString()}</p>
                                                <div className="flex gap-1 md:opacity-0 group-hover:opacity-100 transition-all">
                                                    <button onClick={() => {setEditingId(t.id); setFormData(t); window.scrollTo({top:0, behavior:'smooth'})}} className="p-2 text-indigo-400 hover:bg-indigo-50 rounded-lg text-xs">✎</button>
                                                    <button onClick={() => {if(confirm('ลบรายการ?')) db.ref(`transactions/${t.id}`).remove()}} className="p-2 text-rose-400 hover:bg-rose-50 rounded-lg text-xs">🗑</button>
                                                </div>
                                            </div>
                                        </div>
                                    ))}
                                </div>
                            </div>
                        </div>
                    </div>

                    {showAddAcc && (
                        <div className="fixed inset-0 z-50 flex items-center justify-center bg-rose-200/50 backdrop-blur-md p-6 text-center">
                            <div className="bg-white w-full max-w-sm p-12 rounded-[3.5rem] shadow-2xl border-8 border-white animate-in zoom-in-95 duration-300">
                                <h3 className="text-3xl font-black text-slate-700 mb-8">เพิ่มบัญชีจ้า 🌈</h3>
                                <input autoFocus type="text" value={newAccName} onChange={e => setNewAccName(e.target.value)} placeholder="เช่น กสิกร, เงินเก็บ..." className="w-full px-8 py-5 bg-slate-50 rounded-[2rem] outline-none font-bold text-center mb-10" />
                                <div className="flex gap-4">
                                    <button onClick={() => setShowAddAcc(false)} className="flex-1 py-4 text-slate-300 font-bold">ยกเลิก</button>
                                    <button onClick={handleAddAccount} className="flex-[2] py-4 bg-pink-500 text-white font-black rounded-[2rem] shadow-lg shadow-pink-200">สร้างเลย!</button>
                                </div>
                            </div>
                        </div>
                    )}
                </div>
            );
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
