    <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
    <script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    <script src="https://cdn.sheetjs.com/xlsx-0.20.0/package/dist/xlsx.full.min.js"></script>
    <script src="https://www.gstatic.com/firebasejs/10.7.1/firebase-app-compat.js"></script>
    <script src="https://www.gstatic.com/firebasejs/10.7.1/firebase-database-compat.js"></script>

    <style>
        body { font-family: 'Anuphan', sans-serif; -webkit-tap-highlight-color: transparent; overflow-x: hidden; }
        .custom-scrollbar::-webkit-scrollbar { width: 4px; height: 4px; }

<body class="bg-[#F8FAFC] min-h-screen text-slate-800 p-2 md:p-4">
    <div id="root"></div>

    <script type="text/babel">
        const { useState, useEffect, useMemo } = React;

        const firebaseConfig = {
            apiKey: "AIzaSyALcH0D-G0H6g0yfgLHrxKcYiHZAq7jAGs",
            authDomain: "monney-33caf.firebaseapp.com",
            databaseURL: "https://monney-33caf-default-rtdb.asia-southeast1.firebasedatabase.app",
            projectId: "monney-33caf",
            storageBucket: "monney-33caf.firebasestorage.app",
            messagingSenderId: "709336346002",
            appId: "1:709336346002:web:444bd02c294bd194d6e54b"
        };

        if (!firebase.apps.length) firebase.initializeApp(firebaseConfig);
        const db = firebase.database();

        const App = () => {
            const [accounts, setAccounts] = useState([]);
            const [transactions, setTransactions] = useState([]);
            const [activeAcc, setActiveAcc] = useState('all');
            const [editingId, setEditingId] = useState(null);
            const [showAddAcc, setShowAddAcc] = useState(false);
            const [newAccName, setNewAccName] = useState('');
            const [formData, setFormData] = useState({ description: '', amount: '', type: 'expense', accountId: '', date: new Date().toISOString().split('T')[0] });

            useEffect(() => {
                db.ref('accounts').on('value', snap => {
                    const data = snap.val();
                    if (data) {
                        const list = Object.keys(data).map(k => ({id: k, ...data[k]}));
                        setAccounts(list);
                        if (!formData.accountId && list.length > 0) setFormData(f => ({...f, accountId: list[0].id}));
                    }
                });
                db.ref('transactions').on('value', snap => {
                    const data = snap.val();
                    if (data) {
                        const list = Object.keys(data).map(k => ({
                            id: k, 
                            ...data[k],
                            amount: Number(data[k].amount) || 0 
                        }));
                        setTransactions(list);
                    } else {
                        setTransactions([]);
                    }
                });
            }, []);

            const handleSubmit = (e) => {
                e.preventDefault();
                if (!formData.description || !formData.amount) return;
                const payload = {
                    ...formData,
                    amount: parseFloat(formData.amount),
                    accountId: formData.accountId || (accounts.length > 0 ? accounts[0].id : '')
                };
                if (editingId) db.ref(`transactions/${editingId}`).update(payload);
                else db.ref('transactions').push(payload);
                setEditingId(null);
                setFormData({ ...formData, description: '', amount: '' });
            };

            const stats = useMemo(() => {
                const filtered = transactions.filter(t => activeAcc === 'all' || t.accountId === activeAcc);
                return filtered.reduce((acc, curr) => {
                    const amt = curr.amount;
                    if (curr.type === 'income') { acc.in += amt; acc.bal += amt; }
                    else { acc.out += amt; acc.bal -= amt; }
                    return acc;
                }, { in: 0, out: 0, bal: 0 });
            }, [transactions, activeAcc]);

            const processedTransactions = useMemo(() => {
                const filtered = transactions
                    .filter(t => activeAcc === 'all' || t.accountId === activeAcc)
                    .sort((a, b) => new Date(a.date) - new Date(b.date));
                
                let balance = 0;
                const withBal = filtered.map(t => {
                    balance += (t.type === 'income' ? t.amount : -t.amount);
                    return { ...t, runningBalance: balance };
                });
                return withBal.reverse();
            }, [transactions, activeAcc]);

            const groupedTransactions = useMemo(() => {
                const groups = {};
                processedTransactions.forEach(t => {
                    if (!groups[t.date]) groups[t.date] = [];
                    groups[t.date].push(t);
                });
                return groups;
            }, [processedTransactions]);

            return (
                <div className="max-w-[1200px] mx-auto space-y-4">
                    <header className="flex justify-between items-center bg-white p-3 rounded-xl shadow-sm border border-slate-100">
                        <h1 className="text-lg font-bold text-blue-800 flex items-center gap-2">
                            <i className="fa-solid fa-wallet"></i> บันทึกรายรับรายจ่าย
                        </h1>
                    </header>
                    
                    <div className="grid grid-cols-1 lg:grid-cols-12 gap-4">
                        {/* ฝั่งบันทึก */}
                        <div className="lg:col-span-4">
                            <div className="bg-white p-4 rounded-2xl shadow-sm border border-slate-200">
                                <h2 className="text-md font-bold mb-4 text-slate-700 flex items-center gap-2">
                                    <i className="fa-solid fa-pen-to-square text-blue-500"></i>
                                    {editingId ? 'แก้ไขข้อมูล' : 'บันทึกรายการ'}
                                </h2>
                                <form onSubmit={handleSubmit} className="space-y-3">
                                    <div className="flex bg-slate-100 p-1 rounded-xl">
                                        <button type="button" onClick={() => setFormData({...formData, type: 'income'})} className={`flex-1 py-2 rounded-lg font-bold text-xs transition-all ${formData.type === 'income' ? 'bg-white shadow text-emerald-600' : 'text-slate-400'}`}>รายรับ</button>
                                        <button type="button" onClick={() => setFormData({...formData, type: 'expense'})} className={`flex-1 py-2 rounded-lg font-bold text-xs transition-all ${formData.type === 'expense' ? 'bg-white shadow text-rose-600' : 'text-slate-400'}`}>รายจ่าย</button>
                                    </div>
                                    <div className="grid grid-cols-2 gap-2">
                                        <select value={formData.accountId} onChange={e => setFormData({...formData, accountId: e.target.value})} className="p-2 bg-slate-50 border border-slate-200 rounded-lg text-xs outline-none">
                                            {accounts.map(a => <option key={a.id} value={a.id}>{a.name}</option>)}
                                        </select>
                                        <input type="date" value={formData.date} onChange={e => setFormData({...formData, date: e.target.value})} className="p-2 bg-slate-50 border border-slate-200 rounded-lg text-xs" />
                                    </div>
                                    <input type="text" placeholder="รายละเอียด..." value={formData.description} onChange={e => setFormData({...formData, description: e.target.value})} className="w-full p-2 bg-slate-50 border border-slate-200 rounded-lg text-xs" />
                                    <div className="relative">
                                        <span className="absolute left-3 top-1/2 -translate-y-1/2 font-bold text-blue-300">฿</span>
                                        <input type="number" step="any" placeholder="0.00" value={formData.amount} onChange={e => setFormData({...formData, amount: e.target.value})} className="w-full pl-8 pr-3 py-3 bg-blue-50 border border-blue-100 rounded-xl font-bold text-xl text-right text-blue-700 outline-none" />
                                    </div>
                                    <button type="submit" className="w-full py-3 bg-blue-600 text-white font-bold rounded-xl text-sm shadow-md hover:bg-blue-700 transition-all">บันทึก</button>
                                </form>
                            </div>
                        </div>

                        {/* ฝั่งประวัติ */}
                        <div className="lg:col-span-8 space-y-4">
                            <div className="bg-white p-4 rounded-2xl shadow-sm border border-slate-200">
                                <div className="flex justify-between items-center mb-4">
                                    <h2 className="text-md font-bold text-slate-700">ยอดคงเหลือ</h2>
                                    <button onClick={() => setShowAddAcc(true)} className="text-blue-600 text-[10px] font-bold bg-blue-50 px-2 py-1 rounded-lg">+ เพิ่มบัญชี</button>
                                </div>
                                <div className="grid grid-cols-2 md:grid-cols-4 gap-2 mb-4">
                                    {accounts.map(acc => {
                                        const bal = transactions.filter(t => t.accountId === acc.id).reduce((s, t) => s + (t.type === 'income' ? t.amount : -t.amount), 0);
                                        return (
                                            <div key={acc.id} onClick={() => setActiveAcc(acc.id)} className={`p-3 rounded-xl border transition-all cursor-pointer ${activeAcc === acc.id ? 'border-blue-500 bg-blue-50' : 'border-slate-100 bg-slate-50'}`}>
                                                <p className="font-bold text-slate-500 text-[10px] truncate">{acc.name}</p>
                                                <p className={`text-sm font-bold ${bal >= 0 ? 'text-emerald-600' : 'text-rose-600'}`}>฿{bal.toLocaleString(undefined, {minimumFractionDigits: 2})}</p>
                                            </div>
                                        );
                                    })}
                                </div>
                                <div className="flex justify-between items-center pt-3 border-t">
                                    <p className="text-lg font-bold">รวม: ฿{stats.bal.toLocaleString(undefined, {minimumFractionDigits: 2})}</p>
                                    <button onClick={() => setActiveAcc('all')} className="text-xs font-bold text-slate-400">ดูทั้งหมด</button>
                                </div>
                            </div>

                            <div className="bg-white p-4 rounded-2xl shadow-sm border border-slate-200">
                                <h2 className="text-md font-bold mb-4 text-slate-700">ประวัติรายการ</h2>
                                <div className="space-y-4">
                                    {Object.keys(groupedTransactions).sort((a,b) => new Date(b)-new Date(a)).map(date => (
                                        <div key={date}>
                                            <p className="text-[10px] font-bold text-slate-400 border-b pb-1 mb-2">{date}</p>
                                            <div className="space-y-2">
                                                {groupedTransactions[date].map(t => (
                                                    <div key={t.id} className="grid grid-cols-12 items-center py-2 px-1 border-b border-slate-50 last:border-0 hover:bg-slate-50 rounded-lg transition-all group">
                                                        {/* รายการ (35%) */}
                                                        <div className="col-span-4 min-w-0 pr-1">
                                                            <p className="text-[9px] font-bold text-rose-500 uppercase">รายการ</p>
                                                            <p className="font-bold text-xs text-slate-700 truncate">{t.description}</p>
                                                        </div>

                                                        {/* บัญชี (35%) */}
                                                        <div className="col-span-2 text-center">
                                                            <p className="text-[9px] font-bold text-blue-400 uppercase">บัญชี</p>
                                                            <p className="text-[10px] text-slate-400 truncate">{accounts.find(a => a.id === t.accountId)?.name || '-'}</p>
                                                        </div>

                                                        {/* จำนวนเงิน (35%) */}
                                                        <div className="col-span-3 text-right pr-2">
                                                            <p className="text-[9px] font-bold text-emerald-500 uppercase">เงิน รับ/จ่าย</p>
                                                            <p className={`font-bold text-xs ${t.type === 'income' ? 'text-emerald-600' : 'text-rose-600'}`}>
                                                                {t.type === 'income' ? '+' : '-'}฿{t.amount.toLocaleString(undefined, {minimumFractionDigits: 2})}
                                                            </p>
                                                        </div>

                                                        {/* ยอดคงเหลือ (35%) */}
                                                        <div className="col-span-3 text-right pl-2 border-l border-slate-100 relative">
                                                            <p className="text-[9px] font-bold text-slate-400 uppercase">คงเหลือ</p>
                                                            <p className="font-bold text-xs text-slate-800">฿{t.runningBalance.toLocaleString(undefined, {minimumFractionDigits: 2})}</p>
                                                            <div className="absolute right-0 -top-1 hidden group-hover:flex gap-1 bg-white p-1 rounded-lg shadow-sm border">
                                                                <button onClick={() => {setEditingId(t.id); setFormData(t);}} className="text-blue-500 p-1"><i className="fa-solid fa-pen text-[10px]"></i></button>
                                                                <button onClick={() => { if(confirm('ลบ?')) db.ref(`transactions/${t.id}`).remove(); }} className="text-rose-500 p-1"><i className="fa-solid fa-trash text-[10px]"></i></button>
                                                            </div>
                                                        </div>
                                                    </div>
                                                ))}
                                            </div>
                                        </div>
                                    ))}
                                </div>
                            </div>
                        </div>
                    </div>

                    {showAddAcc && (
                        <div className="fixed inset-0 z-50 flex items-center justify-center bg-slate-900/40 backdrop-blur-sm p-4 text-center">
                            <div className="bg-white w-full max-w-xs p-6 rounded-2xl">
                                <h3 className="text-md font-bold mb-4">เพิ่มบัญชี</h3>
                                <input autoFocus type="text" value={newAccName} onChange={e => setNewAccName(e.target.value)} placeholder="ชื่อบัญชี..." className="w-full px-3 py-3 bg-slate-50 border rounded-xl font-bold text-center mb-4 text-sm" />
                                <div className="flex gap-2">
                                    <button onClick={() => setShowAddAcc(false)} className="flex-1 py-2 text-slate-400 font-bold text-xs">ยกเลิก</button>
                                    <button onClick={() => { if(newAccName) { db.ref('accounts').push({name: newAccName}); setNewAccName(''); setShowAddAcc(false); } }} className="flex-[2] py-2 bg-blue-600 text-white font-bold rounded-xl text-xs shadow-md">บันทึก</button>
                                </div>
                            </div>
                        </div>
                    )}
                </div>
            );
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
