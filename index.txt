    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    <script src="https://cdn.sheetjs.com/xlsx-0.20.0/package/dist/xlsx.full.min.js"></script>
    <script src="https://www.gstatic.com/firebasejs/10.7.1/firebase-app-compat.js"></script>
    <script src="https://www.gstatic.com/firebasejs/10.7.1/firebase-database-compat.js"></script>

        body { font-family: 'Anuphan', sans-serif; -webkit-tap-highlight-color: transparent; }
        .custom-scrollbar::-webkit-scrollbar { width: 4px; height: 4px; }

<body class="bg-[#F1F5F9] min-h-screen text-slate-800 p-2 md:p-6">
    <div id="root"></div>

    <script type="text/babel">
        const { useState, useEffect, useMemo } = React;

        const firebaseConfig = {
            apiKey: "AIzaSyALcH0D-G0H6g0yfgLHrxKcYiHZAq7jAGs",
            authDomain: "monney-33caf.firebaseapp.com",
            databaseURL: "https://monney-33caf-default-rtdb.asia-southeast1.firebasedatabase.app",
            projectId: "monney-33caf",
            storageBucket: "monney-33caf.firebasestorage.app",
            messagingSenderId: "709336346002",
            appId: "1:709336346002:web:444bd02c294bd194d6e54b"
        };

        if (!firebase.apps.length) firebase.initializeApp(firebaseConfig);
        const db = firebase.database();

        const App = () => {
            const [accounts, setAccounts] = useState([]);
            const [transactions, setTransactions] = useState([]);
            const [activeAcc, setActiveAcc] = useState('all');
            const [editingId, setEditingId] = useState(null);
            const [showAddAcc, setShowAddAcc] = useState(false);
            const [newAccName, setNewAccName] = useState('');
            const [formData, setFormData] = useState({ description: '', amount: '', type: 'expense', accountId: '', date: new Date().toISOString().split('T')[0] });

            useEffect(() => {
                db.ref('accounts').on('value', snap => {
                    const data = snap.val();
                    if (data) {
                        const list = Object.keys(data).map(k => ({id: k, ...data[k]}));
                        setAccounts(list);
                        if (!formData.accountId && list.length > 0) setFormData(f => ({...f, accountId: list[0].id}));
                    } else {
                        db.ref('accounts').push({name: 'บัญชีหลัก 💳'});
                    }
                });
                db.ref('transactions').on('value', snap => {
                    const data = snap.val();
                    if (data) setTransactions(Object.keys(data).map(k => ({id: k, ...data[k]})).reverse());
                });
            }, []);

            const exportExcel = () => {
                const wb = XLSX.utils.book_new();
                const summary = accounts.map(acc => {
                    const ts = transactions.filter(t => t.accountId === acc.id);
                    const inc = ts.filter(t => t.type === 'income').reduce((s, t) => s + t.amount, 0);
                    const exp = ts.filter(t => t.type === 'expense').reduce((s, t) => s + t.amount, 0);
                    return { 'บัญชี': acc.name, 'รายรับ': inc, 'รายจ่าย': exp, 'คงเหลือ': inc - exp };
                });
                XLSX.utils.book_append_sheet(wb, XLSX.utils.json_to_sheet(summary), "สรุปยอด");
                accounts.forEach(acc => {
                    const ts = transactions.filter(t => t.accountId === acc.id).sort((a,b) => new Date(a.date) - new Date(b.date));
                    if(ts.length > 0) {
                        let bal = 0;
                        const data = ts.map(t => {
                            bal += (t.type === 'income' ? t.amount : -t.amount);
                            return { 'วันที่': t.date, 'รายการ': t.description, 'รายรับ': t.type === 'income' ? t.amount : 0, 'รายจ่าย': t.type === 'expense' ? t.amount : 0, 'ยอดสะสม': bal };
                        });
                        XLSX.utils.book_append_sheet(wb, XLSX.utils.json_to_sheet(data), acc.name.substring(0, 30));
                    }
                });
                XLSX.writeFile(wb, `Wallet_Report.xlsx`);
            };

            const handleSubmit = (e) => {
                e.preventDefault();
                if (!formData.description || !formData.amount) return;
                const payload = {...formData, amount: parseFloat(formData.amount)};
                if (editingId) db.ref(`transactions/${editingId}`).update(payload);
                else db.ref('transactions').push(payload);
                setEditingId(null);
                setFormData({...formData, description: '', amount: ''});
            };

            const stats = useMemo(() => {
                const filtered = transactions.filter(t => activeAcc === 'all' || t.accountId === activeAcc);
                return filtered.reduce((acc, curr) => {
                    if (curr.type === 'income') { acc.in += curr.amount; acc.bal += curr.amount; }
                    else { acc.out += curr.amount; acc.bal -= curr.amount; }
                    return acc;
                }, { in: 0, out: 0, bal: 0 });
            }, [transactions, activeAcc]);

            const groupedTransactions = useMemo(() => {
                const filtered = transactions.filter(t => activeAcc === 'all' || t.accountId === activeAcc);
                const groups = {};
                filtered.forEach(t => {
                    if (!groups[t.date]) groups[t.date] = [];
                    groups[t.date].push(t);
                });
                return groups;
            }, [transactions, activeAcc]);

            return (
                <div className="max-w-7xl mx-auto space-y-6">
                    <header className="flex justify-between items-center bg-white p-4 rounded-2xl shadow-sm border border-slate-100">
                        <div className="flex items-center gap-3">
                            <div className="bg-blue-600 w-10 h-10 rounded-xl flex items-center justify-center text-white shadow-lg shadow-blue-200">
                                <i className="fa-solid fa-wallet"></i>
                            </div>
                            <h1 className="text-xl font-bold bg-clip-text text-transparent bg-gradient-to-r from-blue-800 to-blue-600">SMART WALLET</h1>
                        </div>
                        <button onClick={exportExcel} className="bg-emerald-500 hover:bg-emerald-600 text-white px-4 py-2 rounded-xl text-xs font-bold transition-all flex items-center gap-2">
                            <i className="fa-solid fa-file-excel"></i> EXCEL
                        </button>
                    </header>
                    
                    <div className="grid grid-cols-1 lg:grid-cols-12 gap-6">
                        
                        {/* ฝั่งซ้าย: ฟอร์มบันทึกข้อมูล */}
                        <div className="lg:col-span-5">
                            <div className="bg-white p-6 rounded-3xl shadow-sm border border-slate-200 h-full">
                                <div className="flex items-center gap-2 mb-6">
                                    <i className="fa-solid fa-pen-to-square text-blue-600"></i>
                                    <h2 className="text-lg font-bold text-slate-700">{editingId ? 'แก้ไขรายการ' : 'บันทึกรายการ'}</h2>
                                </div>
                                <form onSubmit={handleSubmit} className="space-y-4">
                                    <div className="flex bg-slate-100 p-1 rounded-2xl">
                                        <button type="button" onClick={() => setFormData({...formData, type: 'income'})} className={`flex-1 py-3 rounded-xl font-bold text-sm transition-all ${formData.type === 'income' ? 'bg-white shadow-sm text-emerald-600' : 'text-slate-500 hover:text-slate-700'}`}>
                                            <i className="fa-solid fa-arrow-down-long mr-2"></i>รายรับ
                                        </button>
                                        <button type="button" onClick={() => setFormData({...formData, type: 'expense'})} className={`flex-1 py-3 rounded-xl font-bold text-sm transition-all ${formData.type === 'expense' ? 'bg-white shadow-sm text-rose-600' : 'text-slate-500 hover:text-slate-700'}`}>
                                            <i className="fa-solid fa-arrow-up-long mr-2"></i>รายจ่าย
                                        </button>
                                    </div>
                                    <div className="grid grid-cols-2 gap-3">
                                        <div className="relative">
                                            <i className="fa-solid fa-building-columns absolute left-3 top-3.5 text-slate-400 text-xs"></i>
                                            <select value={formData.accountId} onChange={e => setFormData({...formData, accountId: e.target.value})} className="w-full pl-9 pr-3 py-3 bg-slate-50 border border-slate-200 rounded-xl text-sm focus:ring-2 focus:ring-blue-100 focus:border-blue-400 outline-none transition-all">
                                                {accounts.map(a => <option key={a.id} value={a.id}>{a.name}</option>)}
                                            </select>
                                        </div>
                                        <div className="relative">
                                            <i className="fa-solid fa-calendar-days absolute left-3 top-3.5 text-slate-400 text-xs"></i>
                                            <input type="date" value={formData.date} onChange={e => setFormData({...formData, date: e.target.value})} className="w-full pl-9 pr-3 py-3 bg-slate-50 border border-slate-200 rounded-xl text-sm focus:ring-2 focus:ring-blue-100 focus:border-blue-400 outline-none transition-all" />
                                        </div>
                                    </div>
                                    <div className="relative">
                                        <i className="fa-solid fa-tag absolute left-3 top-3.5 text-slate-400 text-xs"></i>
                                        <input type="text" placeholder="ระบุรายละเอียด..." value={formData.description} onChange={e => setFormData({...formData, description: e.target.value})} className="w-full pl-9 pr-3 py-3 bg-slate-50 border border-slate-200 rounded-xl text-sm focus:ring-2 focus:ring-blue-100 focus:border-blue-400 outline-none" />
                                    </div>
                                    <div className="relative">
                                        <span className="absolute left-4 top-1/2 -translate-y-1/2 font-bold text-blue-400 text-2xl">฿</span>
                                        <input type="number" placeholder="0.00" value={formData.amount} onChange={e => setFormData({...formData, amount: e.target.value})} className="w-full pl-12 pr-4 py-5 bg-blue-50 border-2 border-blue-100 rounded-2xl font-bold text-3xl text-right text-blue-700 focus:ring-4 focus:ring-blue-100 outline-none transition-all" />
                                    </div>
                                    <button type="submit" className="w-full py-4 bg-gradient-to-r from-blue-600 to-blue-700 text-white font-bold rounded-2xl shadow-lg shadow-blue-100 hover:scale-[1.02] active:scale-95 transition-all">
                                        {editingId ? <i className="fa-solid fa-check mr-2"></i> : <i className="fa-solid fa-plus mr-2"></i>}
                                        {editingId ? 'บันทึกการแก้ไข' : 'บันทึกรายการ'}
                                    </button>
                                </form>
                            </div>
                        </div>

                        {/* ฝั่งขวา: แสดงยอดเงินรายบัญชี */}
                        <div className="lg:col-span-7">
                            <div className="bg-white p-6 rounded-3xl shadow-sm border border-slate-200 h-full flex flex-col">
                                <div className="flex justify-between items-center mb-6">
                                    <div className="flex items-center gap-2">
                                        <i className="fa-solid fa-vault text-emerald-500"></i>
                                        <h2 className="text-lg font-bold text-slate-700">ยอดเงินในบัญชี</h2>
                                    </div>
                                    <button onClick={() => setShowAddAcc(true)} className="bg-slate-800 hover:bg-slate-900 text-white px-4 py-2 rounded-xl text-xs font-bold transition-all flex items-center gap-2">
                                        <i className="fa-solid fa-plus"></i> เพิ่มบัญชี
                                    </button>
                                </div>
                                
                                <div className="grid grid-cols-1 md:grid-cols-2 gap-3 overflow-y-auto pr-2 custom-scrollbar flex-grow" style={{maxHeight: '300px'}}>
                                    {accounts.map(acc => {
                                        const accTs = transactions.filter(t => t.accountId === acc.id);
                                        const balance = accTs.reduce((sum, t) => sum + (t.type === 'income' ? t.amount : -t.amount), 0);
                                        return (
                                            <div key={acc.id} onClick={() => setActiveAcc(acc.id)} className={`p-4 rounded-2xl border cursor-pointer transition-all ${activeAcc === acc.id ? 'border-blue-500 bg-blue-50/50 shadow-sm ring-1 ring-blue-500' : 'border-slate-100 bg-slate-50 hover:border-blue-200'}`}>
                                                <div className="flex justify-between items-start">
                                                    <div className="flex-1">
                                                        <div className="flex items-center gap-2 mb-1">
                                                            <i className={`fa-solid fa-credit-card text-[10px] ${activeAcc === acc.id ? 'text-blue-500' : 'text-slate-400'}`}></i>
                                                            <p className="font-bold text-slate-700 text-sm truncate">{acc.name}</p>
                                                        </div>
                                                        <p className={`text-xl font-bold ${balance >= 0 ? 'text-slate-800' : 'text-rose-600'}`}>
                                                            <span className="text-sm font-normal mr-1">฿</span>{balance.toLocaleString()}
                                                        </p>
                                                    </div>
                                                    <button onClick={(e) => {e.stopPropagation(); if(confirm('ลบบัญชีนี้?')) db.ref(`accounts/${acc.id}`).remove()}} className="text-slate-300 hover:text-rose-500 transition-colors">
                                                        <i className="fa-solid fa-circle-xmark"></i>
                                                    </button>
                                                </div>
                                            </div>
                                        );
                                    })}
                                </div>

                                <div className="mt-6 pt-5 border-t border-slate-100 flex justify-between items-center">
                                    <div>
                                        <p className="text-[10px] font-bold text-slate-400 uppercase tracking-wider mb-1">ยอดรวมทั้งหมด</p>
                                        <p className="text-3xl font-bold text-blue-900 leading-none">
                                            <span className="text-lg font-normal mr-1">฿</span>{stats.bal.toLocaleString()}
                                        </p>
                                    </div>
                                    <button onClick={() => setActiveAcc('all')} className={`px-5 py-2.5 rounded-xl text-xs font-bold transition-all ${activeAcc === 'all' ? 'bg-blue-600 text-white shadow-md' : 'bg-slate-100 text-slate-500 hover:bg-slate-200'}`}>
                                        แสดงทุกบัญชี
                                    </button>
                                </div>
                            </div>
                        </div>

                    {/* ส่วนล่าง: ประวัติรายการ */}

                        <div className="flex items-center gap-2 mb-6">
                            <i className="fa-solid fa-clock-rotate-left text-blue-500"></i>
                            <h2 className="text-lg font-bold text-slate-700">ประวัติรายการ {activeAcc !== 'all' && <span className="text-blue-600 ml-1">({accounts.find(a => a.id === activeAcc)?.name})</span>}</h2>
                        </div>
                        <div className="space-y-8">
                            {Object.keys(groupedTransactions).sort((a,b) => new Date(b)-new Date(a)).map(date => (
                                <div key={date} className="relative">
                                    <p className="sticky top-0 bg-white py-2 text-[11px] font-bold text-slate-400 border-b border-slate-50 mb-3 flex items-center gap-2">
                                        <i className="fa-regular fa-calendar"></i> {date}
                                    </p>
                                    <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
                                        {groupedTransactions[date].map(t => (
                                            <div key={t.id} className="flex justify-between items-center p-4 bg-slate-50/50 rounded-2xl group hover:bg-white border border-transparent hover:border-slate-200 transition-all">
                                                <div className="flex-1 min-w-0 pr-4">
                                                    <p className="font-bold text-sm text-slate-700 truncate mb-1">{t.description}</p>
                                                    <span className="inline-block px-2 py-0.5 bg-blue-100 text-blue-600 rounded text-[9px] font-bold uppercase">
                                                        {accounts.find(a => a.id === t.accountId)?.name}
                                                    </span>
                                                </div>
                                                <div className="flex items-center gap-4">
                                                    <p className={`font-bold text-sm ${t.type === 'income' ? 'text-emerald-600' : 'text-slate-800'}`}>
                                                        {t.type === 'income' ? '+' : '-'}฿{t.amount.toLocaleString()}
                                                    </p>
                                                    <div className="flex gap-2 opacity-0 group-hover:opacity-100 transition-all">
                                                        <button onClick={() => {setEditingId(t.id); setFormData(t); window.scrollTo(0,0)}} className="w-8 h-8 rounded-full bg-blue-50 text-blue-600 flex items-center justify-center hover:bg-blue-600 hover:text-white transition-all shadow-sm">
                                                            <i className="fa-solid fa-pencil text-[10px]"></i>
                                                        </button>
                                                        <button onClick={() => {if(confirm('ลบรายการ?')) db.ref(`transactions/${t.id}`).remove()}} className="w-8 h-8 rounded-full bg-rose-50 text-rose-500 flex items-center justify-center hover:bg-rose-600 hover:text-white transition-all shadow-sm">
                                                            <i className="fa-solid fa-trash-can text-[10px]"></i>
                                                        </button>
                                                    </div>
                                                </div>
                                            </div>
                                        ))}
                                    </div>
                                </div>
                            ))}
                        </div>
                    </div>

                    {/* Modal เพิ่มบัญชี */}
                    {showAddAcc && (
                        <div className="fixed inset-0 z-50 flex items-center justify-center bg-slate-900/60 backdrop-blur-sm p-4">
                            <div className="bg-white w-full max-w-xs p-8 rounded-3xl shadow-2xl scale-in-center">
                                <div className="w-16 h-16 bg-blue-50 rounded-full flex items-center justify-center text-blue-600 text-2xl mx-auto mb-4">
                                    <i className="fa-solid fa-plus"></i>
                                </div>
                                <h3 className="text-lg font-bold text-center mb-6 text-slate-700">เพิ่มกระเป๋าเงินใหม่</h3>
                                <input autoFocus type="text" value={newAccName} onChange={e => setNewAccName(e.target.value)} placeholder="เช่น กสิกร, บัญชีเงินเก็บ..." className="w-full px-4 py-4 bg-slate-50 border border-slate-200 rounded-2xl font-bold text-center mb-6 focus:ring-2 focus:ring-blue-100 focus:border-blue-400 outline-none" />
                                <div className="flex gap-3">
                                    <button onClick={() => setShowAddAcc(false)} className="flex-1 py-3 text-slate-400 font-bold text-sm hover:bg-slate-50 rounded-xl transition-all">ยกเลิก</button>
                                    <button onClick={() => {if(newAccName){db.ref('accounts').push({name: newAccName}); setNewAccName(''); setShowAddAcc(false);}}} className="flex-[2] py-3 bg-blue-600 text-white font-bold rounded-xl text-sm shadow-lg shadow-blue-100 hover:bg-blue-700 transition-all">บันทึก</button>
                                </div>
                            </div>
                        </div>
                    )}
                </div>
            );
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
