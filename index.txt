<html lang="th">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>My Sweet Finance Diary ✨</title>
    <!-- โหลด Tailwind CSS สำหรับการตกแต่ง -->
    <script src="https://cdn.tailwindcss.com"></script>
    <link href="https://fonts.googleapis.com/css2?family=Anuphan:wght@100;400;700&display=swap" rel="stylesheet">
    <!-- โหลด React และ Babel -->
    <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
    <script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    <!-- โหลด SheetJS สำหรับ Excel -->
    <script src="https://cdn.sheetjs.com/xlsx-0.20.0/package/dist/xlsx.full.min.js"></script>
    
    <style>
        body { font-family: 'Anuphan', sans-serif; overflow-x: hidden; }
        .custom-scrollbar::-webkit-scrollbar { width: 8px; }
        .custom-scrollbar::-webkit-scrollbar-track { background: transparent; }
        .custom-scrollbar::-webkit-scrollbar-thumb { background: #FFC0CB; border-radius: 20px; border: 3px solid white; }
        @keyframes float { 0%, 100% { transform: translateY(0); } 50% { transform: translateY(-12px); } }
        .float-anim { animation: float 4s ease-in-out infinite; }
    </style>
</head>
<body class="bg-gradient-to-br from-[#FFF0F5] via-[#F8FAFF] to-[#F0FFF4] min-h-screen">
    <div id="root"></div>

    <script type="text/babel">
        const { useState, useEffect, useMemo } = React;

        // --- ไอคอนแบบ SVG (เพื่อให้โค้ดไม่พังเวลาเน็ตช้า) ---
        const Icon = ({ name, size = 24, className = "" }) => {
            const icons = {
                wallet: <path d="M19 7V4a1 1 0 0 0-1-1H5a2 2 0 0 0 0 4 2 2 0 0 0 0 4h13a1 1 0 0 0 1-1v-2a1 1 0 0 0-1-1H3" />,
                plus: <path d="M12 5v14M5 12h14" />,
                trash: <path d="M3 6h18M19 6v14c0 1-1 2-2 2H7c-1 0-2-1-2-2V6M8 6V4c0-1 1-2 2-2h4c1 0 2 1 2 2v2" />,
                pencil: <path d="M17 3a2.85 2.83 0 1 1 4 4L7.5 20.5 2 22l1.5-5.5Z" />,
                flower: <path d="M12 7.5a4.5 4.5 0 1 1 4.5 4.5M12 7.5A4.5 4.5 0 1 0 7.5 12M12 7.5V12m0 0a4.5 4.5 0 1 0 4.5 4.5M12 12a4.5 4.5 0 1 1-4.5 4.5" />,
                download: <path d="M21 15v4a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2v-4M7 10l5 5 5-5M12 15V3" />,
                chart: <path d="M12 20V10M18 20V4M6 20v-4" />
            };
            return (
                <svg width={size} height={size} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className={className}>
                    {icons[name] || <circle cx="12" cy="12" r="10" />}
                </svg>
            );
        };

        const App = () => {
            const [accounts, setAccounts] = useState(() => {
                const saved = localStorage.getItem('diary_accounts_v4');
                return saved ? JSON.parse(saved) : [{ id: '1', name: 'เงินออม', color: 'from-pink-400 to-rose-300' }];
            });
            const [transactions, setTransactions] = useState(() => {
                const saved = localStorage.getItem('diary_trans_v4');
                return saved ? JSON.parse(saved) : [];
            });
            const [activeAcc, setActiveAcc] = useState('all');
            const [showAddAcc, setShowAddAcc] = useState(false);
            const [newAccName, setNewAccName] = useState('');
            const [editingId, setEditingId] = useState(null);
            const [formData, setFormData] = useState({ description: '', amount: '', type: 'expense', accountId: accounts[0].id, date: new Date().toISOString().split('T')[0] });

            useEffect(() => localStorage.setItem('diary_accounts_v4', JSON.stringify(accounts)), [accounts]);
            useEffect(() => localStorage.setItem('diary_trans_v4', JSON.stringify(transactions)), [transactions]);

            const handleAddAccount = () => {
                if (!newAccName.trim()) return;
                const colors = ['from-pink-400 to-rose-300', 'from-sky-400 to-blue-300', 'from-amber-400 to-orange-300', 'from-indigo-400 to-purple-300'];
                setAccounts([...accounts, { id: Date.now().toString(), name: newAccName, color: colors[accounts.length % colors.length] }]);
                setNewAccName('');
                setShowAddAcc(false);
            };

            const handleSubmit = (e) => {
                e.preventDefault();
                if (!formData.description || !formData.amount) return;
                if (editingId) {
                    setTransactions(transactions.map(t => t.id === editingId ? { ...t, ...formData, amount: parseFloat(formData.amount) } : t));
                    setEditingId(null);
                } else {
                    setTransactions([{ id: Date.now(), ...formData, amount: parseFloat(formData.amount) }, ...transactions]);
                }
                setFormData({ ...formData, description: '', amount: '' });
            };

            const stats = useMemo(() => {
                const filtered = transactions.filter(t => activeAcc === 'all' || t.accountId === activeAcc);
                return filtered.reduce((acc, curr) => {
                    if (curr.type === 'income') { acc.in += curr.amount; acc.bal += curr.amount; }
                    else { acc.out += curr.amount; acc.bal -= curr.amount; }
                    return acc;
                }, { in: 0, out: 0, bal: 0 });
            }, [transactions, activeAcc]);

            const exportExcel = () => {
                const wb = XLSX.utils.book_new();
                const data = transactions.map(t => ({
                    'บัญชี': accounts.find(a => a.id === t.accountId)?.name,
                    'วันที่': t.date, 'รายการ': t.description, 'ประเภท': t.type === 'income' ? 'รับ' : 'จ่าย', 'จำนวน': t.amount
                }));
                XLSX.utils.book_append_sheet(wb, XLSX.utils.json_to_sheet(data), "Summary");
                XLSX.writeFile(wb, "My_Finance.xlsx");
            };

            return (
                <div className="max-w-6xl mx-auto px-4 py-12 flex flex-col items-center">
                    <div className="float-anim bg-gradient-to-tr from-pink-500 to-orange-300 p-6 rounded-[3rem] text-white shadow-2xl mb-6">
                        <Icon name="flower" size={48} />
                    </div>
                    <h1 className="text-4xl font-black text-rose-500 mb-2">Diary Finance</h1>
                    <p className="text-slate-400 font-bold mb-10 italic">บันทึกความสุขและการออม ✨</p>

                    {/* ตัวเลือกบัญชี */}
                    <div className="flex flex-wrap justify-center gap-3 bg-white/50 backdrop-blur-md p-4 rounded-[2.5rem] shadow-sm mb-12 w-full max-w-4xl">
                        <button onClick={() => setActiveAcc('all')} className={`px-6 py-3 rounded-2xl font-black text-xs transition-all ${activeAcc === 'all' ? 'bg-pink-500 text-white shadow-lg' : 'text-pink-400'}`}>ทั้งหมด 🌈</button>
                        {accounts.map(a => (
                            <button key={a.id} onClick={() => setActiveAcc(a.id)} className={`flex items-center gap-2 px-6 py-3 rounded-2xl font-black text-xs transition-all ${activeAcc === a.id ? 'bg-white shadow-md text-slate-800 scale-105' : 'text-slate-400'}`}>
                                <div className={`w-3 h-3 rounded-full bg-gradient-to-br ${a.color}`}></div>{a.name}
                            </button>
                        ))}
                        <button onClick={() => setShowAddAcc(true)} className="w-10 h-10 bg-white text-pink-500 rounded-2xl shadow-sm flex items-center justify-center font-bold">+</button>
                    </div>

                    {/* แผงสรุปยอด */}
                    <div className="grid grid-cols-1 md:grid-cols-3 gap-6 w-full mb-12">
                        <div className="bg-white/80 p-8 rounded-[3rem] shadow-xl text-center border-b-8 border-pink-100">
                            <p className="text-[10px] font-black text-slate-300 uppercase mb-2">คงเหลือรวม</p>
                            <p className="text-4xl font-black text-slate-800">฿{stats.bal.toLocaleString()}</p>
                        </div>
                        <div className="bg-emerald-400 p-8 rounded-[3rem] shadow-xl text-center text-white">
                            <p className="text-emerald-100 text-[10px] font-black uppercase mb-2">รายรับ</p>
                            <p className="text-4xl font-black">฿{stats.in.toLocaleString()}</p>
                        </div>
                        <div className="bg-rose-400 p-8 rounded-[3rem] shadow-xl text-center text-white">
                            <p className="text-rose-100 text-[10px] font-black uppercase mb-2">รายจ่าย</p>
                            <p className="text-4xl font-black">฿{stats.out.toLocaleString()}</p>
                        </div>
                    </div>

                    <div className="grid grid-cols-1 lg:grid-cols-12 gap-10 w-full items-start">
                        {/* ฟอร์มจดบันทึก */}
                        <div className="lg:col-span-5 bg-white p-10 rounded-[3.5rem] shadow-2xl border-4 border-white relative">
                            <h3 className="text-2xl font-black text-slate-800 mb-8 flex items-center gap-3">
                                {editingId ? '🪄 แก้ไขรายการ' : '📔 จดรายการใหม่'}
                            </h3>
                            <form onSubmit={handleSubmit} className="space-y-6">
                                <div className="flex p-2 bg-slate-50 rounded-[2rem]">
                                    <button type="button" onClick={() => setFormData({...formData, type: 'income'})} className={`flex-1 py-4 rounded-2xl text-sm font-black transition-all ${formData.type === 'income' ? 'bg-white text-emerald-500 shadow-md' : 'text-slate-300'}`}>ได้เงินมา 🍰</button>
                                    <button type="button" onClick={() => setFormData({...formData, type: 'expense'})} className={`flex-1 py-4 rounded-2xl text-sm font-black transition-all ${formData.type === 'expense' ? 'bg-white text-rose-500 shadow-md' : 'text-slate-300'}`}>ใช้เงินไป 🎈</button>
                                </div>
                                <select value={formData.accountId} onChange={e => setFormData({...formData, accountId: e.target.value})} className="w-full px-6 py-4 bg-slate-50 rounded-2xl font-bold text-sm outline-none border-none">
                                    {accounts.map(a => <option key={a.id} value={a.id}>{a.name}</option>)}
                                </select>
                                <input type="text" placeholder="รายการอะไรเอ่ย?" value={formData.description} onChange={e => setFormData({...formData, description: e.target.value})} className="w-full px-8 py-5 bg-slate-50 rounded-[2rem] outline-none font-bold shadow-inner" />
                                <input type="number" placeholder="กี่บาทจ๊ะ?" value={formData.amount} onChange={e => setFormData({...formData, amount: e.target.value})} className="w-full px-8 py-5 bg-slate-50 rounded-[2rem] outline-none text-2xl font-black text-center shadow-inner" />
                                <button type="submit" className="w-full py-6 rounded-[2.5rem] text-white font-black text-lg bg-gradient-to-r from-pink-500 to-rose-400 shadow-xl active:scale-95 transition-all">บันทึกเลยจ้า ✨</button>
                                <button type="button" onClick={exportExcel} className="w-full py-4 text-emerald-500 font-black text-xs border-2 border-emerald-100 rounded-[2rem] hover:bg-emerald-50 transition-all">ส่งออก EXCEL 📑</button>
                            </form>
                        </div>

                        {/* รายการที่บันทึกไว้ */}
                        <div className="lg:col-span-7 bg-white/40 backdrop-blur-xl rounded-[4rem] border-4 border-white shadow-2xl p-10 min-h-[500px]">
                            <h3 className="text-2xl font-black text-slate-800 mb-8">ประวัติบันทึก 📒</h3>
                            <div className="space-y-4 max-h-[600px] overflow-y-auto pr-2 custom-scrollbar">
                                {transactions.filter(t => activeAcc === 'all' || t.accountId === activeAcc).map(t => (
                                    <div key={t.id} className="p-6 bg-white/90 rounded-[2.5rem] border-2 border-white shadow-sm flex items-center justify-between group hover:shadow-md transition-all">
                                        <div className="flex items-center gap-4">
                                            <div className={`w-12 h-12 rounded-2xl flex items-center justify-center text-xl ${t.type === 'income' ? 'bg-emerald-50 text-emerald-500' : 'bg-rose-50 text-rose-500'}`}>
                                                {t.type === 'income' ? '🍰' : '🎈'}
                                            </div>
                                            <div>
                                                <p className="font-black text-slate-700">{t.description}</p>
                                                <p className="text-[10px] text-slate-300 font-bold uppercase">{t.date}</p>
                                            </div>
                                        </div>
                                        <div className="flex items-center gap-4">
                                            <p className={`text-xl font-black ${t.type === 'income' ? 'text-emerald-500' : 'text-slate-800'}`}>฿{t.amount.toLocaleString()}</p>
                                            <div className="flex gap-1 opacity-0 group-hover:opacity-100 transition-all">
                                                <button onClick={() => handleEdit(t)} className="p-2 text-indigo-400 hover:bg-indigo-50 rounded-lg"><Icon name="pencil" size={16}/></button>
                                                <button onClick={() => setTransactions(transactions.filter(tr => tr.id !== t.id))} className="p-2 text-rose-400 hover:bg-rose-50 rounded-lg"><Icon name="trash" size={16}/></button>
                                            </div>
                                        </div>
                                    </div>
                                ))}
                                {transactions.length === 0 && <p className="text-center py-20 text-slate-300 font-bold italic">ยังไม่มีข้อมูลเลยน้า...</p>}
                            </div>
                        </div>
                    </div>

                    {/* Modal เพิ่มบัญชี */}
                    {showAddAcc && (
                        <div className="fixed inset-0 z-50 flex items-center justify-center bg-rose-200/50 backdrop-blur-md p-6">
                            <div className="bg-white w-full max-w-sm p-12 rounded-[3.5rem] shadow-2xl text-center border-8 border-white animate-in zoom-in-95 duration-300">
                                <h3 className="text-3xl font-black text-slate-700 mb-8">เพิ่มกระเป๋าจ้า 🌈</h3>
                                <input autoFocus type="text" value={newAccName} onChange={e => setNewAccName(e.target.value)} placeholder="ชื่อบัญชี..." className="w-full px-8 py-5 bg-slate-50 rounded-[2rem] outline-none font-bold text-center mb-10 shadow-inner" />
                                <div className="flex gap-4">
                                    <button onClick={() => setShowAddAcc(false)} className="flex-1 py-4 text-slate-300 font-bold">ยกเลิก</button>
                                    <button onClick={handleAddAccount} className="flex-[2] py-4 bg-pink-500 text-white font-black rounded-[2rem] shadow-lg shadow-pink-200">สร้างเลย!</button>
                                </div>
                            </div>
                        </div>
                    )}
                </div>
            );
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
</body>
</html>
